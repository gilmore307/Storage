import os
import subprocess
from datetime import datetime
import pystray
from pystray import MenuItem as item
from PIL import Image
import time

# ===============================
# 配置
# ===============================
SSH_USER = "root"
SSH_HOST = "100.110.207.125"
LOCAL_PORT = 18789
REMOTE_PORT = 18789
TRAY_ICON_PATH = r"C:\OpenClaw\OpenClaw.jfif"
LOG_DIR = r"C:\OpenClaw\tray\logs"
LOG_PATH = os.path.join(LOG_DIR, "win-node-1-office.log")
TRAY_EXE_NAME = "win-node-1-office"
GATEWAY_TOKEN = "在这里替换成新的GATEWAY_TOKEN"
NODEJS_PATH = r"C:\Program Files\nodejs\node.exe"
OPENCLAW_PATH = r"C:\Users\sunch\AppData\Roaming\npm\node_modules\openclaw\dist\index.js"

# ===============================
# 设置环境变量
# ===============================
env = os.environ.copy()
env["OPENCLAW_GATEWAY_TOKEN"] = GATEWAY_TOKEN
env["OPENCLAW_GATEWAY_URL"] = f"ws://127.0.0.1:{LOCAL_PORT}/"
env["OPENCLAW_ALLOW_INSECURE_PRIVATE_WS"] = "1"

# ===============================
# 日志函数
# ===============================
def ensure_log_dir():
    os.makedirs(LOG_DIR, exist_ok=True)

def write_log(msg: str):
    ensure_log_dir()
    with open(LOG_PATH, "a", encoding="utf-8") as f:
        ts = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        f.write(f"[{ts}] [tray] {msg}\n")
        f.flush()

# ===============================
# PowerShell helper
# ===============================
def _run_powershell(ps_cmd: str):
    return subprocess.run(
        ["powershell", "-NoProfile", "-ExecutionPolicy", "Bypass", "-Command", ps_cmd],
        capture_output=True,
        text=True,
        shell=False
    )

# ===============================
# 杀旧进程
# ===============================
def kill_old_node_processes():
    write_log("Checking for old OpenClaw node processes...")
    ps_cmd = rf"""
$procs = Get-CimInstance Win32_Process | Where-Object {{
    $_.Name -eq 'node.exe' -and
    $_.CommandLine -match [regex]::Escape('{OPENCLAW_PATH}') -and
    $_.CommandLine -match 'node run'
}}
foreach ($p in $procs) {{
    try {{
        Stop-Process -Id $p.ProcessId -Force -ErrorAction Stop
        Write-Output "Killed node PID=$($p.ProcessId)"
    }} catch {{
        Write-Output "Failed to kill node PID=$($p.ProcessId): $($_.Exception.Message)"
    }}
}}
"""
    result = _run_powershell(ps_cmd)
    output = (result.stdout or "").strip()
    err = (result.stderr or "").strip()
    if output:
        write_log(output)
    else:
        write_log("No old OpenClaw node processes found.")
    if err:
        write_log(f"PowerShell stderr: {err}")

def kill_old_ssh_tunnels():
    write_log("Checking for old SSH tunnel processes...")
    ps_cmd = rf"""
$procs = Get-CimInstance Win32_Process | Where-Object {{
    $_.Name -eq 'ssh.exe' -and
    $_.CommandLine -match '-L {LOCAL_PORT}:127\.0\.0\.1:{REMOTE_PORT}' -and
    $_.CommandLine -match '{SSH_USER}@{SSH_HOST}'
}}
foreach ($p in $procs) {{
    try {{
        Stop-Process -Id $p.ProcessId -Force -ErrorAction Stop
        Write-Output "Killed ssh PID=$($p.ProcessId)"
    }} catch {{
        Write-Output "Failed to kill ssh PID=$($p.ProcessId): $($_.Exception.Message)"
    }}
}}
"""
    result = _run_powershell(ps_cmd)
    output = (result.stdout or "").strip()
    err = (result.stderr or "").strip()
    if output:
        write_log(output)
    else:
        write_log("No old SSH tunnel processes found.")
    if err:
        write_log(f"PowerShell stderr: {err}")

# ===============================
# 启动 SSH 隧道后台
# ===============================
def start_ssh_tunnel():
    kill_old_ssh_tunnels()
    write_log(f"Starting SSH tunnel: {SSH_USER}@{SSH_HOST}:{REMOTE_PORT}->{LOCAL_PORT}")
    subprocess.Popen(
        ["ssh", "-N", "-L", f"{LOCAL_PORT}:127.0.0.1:{REMOTE_PORT}", f"{SSH_USER}@{SSH_HOST}"],
        stdin=subprocess.DEVNULL,
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
        shell=False,
        creationflags=subprocess.CREATE_NO_WINDOW | subprocess.DETACHED_PROCESS
    )
    time.sleep(2)
    write_log("SSH tunnel started.")

# ===============================
# 启动 Node 后台
# ===============================
def start_node():
    kill_old_node_processes()
    write_log("Starting OpenClaw Node in background...")
    subprocess.Popen(
        [NODEJS_PATH, OPENCLAW_PATH, "node", "run",
         "--host", "127.0.0.1",
         "--port", str(LOCAL_PORT),
         "--display-name", TRAY_EXE_NAME],
        stdin=subprocess.DEVNULL,
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
        shell=False,
        creationflags=subprocess.CREATE_NO_WINDOW | subprocess.DETACHED_PROCESS,
        env=env
    )
    write_log("Node started.")

# ===============================
# 停止后台
# ===============================
def stop_worker(icon=None, menu_item=None):
    write_log("Stopping old node and SSH tunnel processes...")
    kill_old_node_processes()
    kill_old_ssh_tunnels()
    if icon:
        icon.notify("OpenClaw node stopped", "OpenClaw")

def restart_worker(icon=None, menu_item=None):
    write_log("Restart requested.")
    stop_worker(icon, menu_item)
    time.sleep(1)
    start_ssh_tunnel()
    start_node()
    if icon:
        icon.notify("OpenClaw node restarted", "OpenClaw")

def on_exit(icon=None, menu_item=None):
    write_log("Exit requested.")
    stop_worker(icon, menu_item)
    icon.stop()

# ===============================
# 托盘主程序
# ===============================
def load_icon():
    return Image.open(TRAY_ICON_PATH)

def main():
    write_log("Tray started.")
    start_ssh_tunnel()
    start_node()

    tray_icon = pystray.Icon(
        "openclaw_tray",
        load_icon(),
        "OpenClaw Tray",
        menu=pystray.Menu(
            item("Stop", stop_worker),
            item("Restart", restart_worker),
            item("Exit", on_exit)
        )
    )
    tray_icon.run()

if __name__ == "__main__":
    main()
